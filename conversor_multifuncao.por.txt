programa
{
	funcao inicio()
	{
		inteiro opcao
		real valor
		real resultado

		enquanto(verdadeiro)
		{
			escreva("\n=== CONVERSOR COMPLETO ===\n")
			escreva("1 - Celsius para Fahrenheit\n")
			escreva("2 - Km para Milhas\n")
			escreva("3 - Kg para Libras\n")
			escreva("4 - Real para Dólar (taxa fixa)\n")
			escreva("5 - Sair\n")
			escreva("Escolha: ")
			leia(opcao)

			escolha(opcao)
			{
				caso 1:
					escreva("Digite Celsius: ")
					leia(valor)

					resultado = (valor * 9/5) + 32
					escreva("Fahrenheit: ", resultado)
					pare

				caso 2:
					escreva("Digite Km: ")
					leia(valor)

					resultado = valor * 0.621
					escreva("Milhas: ", resultado)
					pare

				caso 3:
					escreva("Digite Kg: ")
					leia(valor)

					resultado = valor * 2.205
					escreva("Libras: ", resultado)
					pare

				caso 4:
					escreva("Digite valor em Reais: ")
					leia(valor)

					resultado = valor / 5.0
					escreva("Dólares: ", resultado)
					pare

				caso 5:
					escreva("Encerrando...\n")
					pare

				caso contrario:
					escreva("Opção inválida!\n")
			}
		}
	}
}
