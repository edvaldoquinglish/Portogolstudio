programa
{
	funcao inicio()
	{
		cadeia nome
		real preco
		inteiro quantidade
		real total

		escreva("=== CADASTRO DE PRODUTOS ===\n\n")

		escreva("Nome do produto: ")
		leia(nome)

		escreva("Preço do produto: ")
		leia(preco)

		escreva("Quantidade em estoque: ")
		leia(quantidade)

		total = preco * quantidade

		escreva("\n=== RELATÓRIO ===\n")
		escreva("Produto: ", nome, "\n")
		escreva("Preço: ", preco, "\n")
		escreva("Quantidade: ", quantidade, "\n")
		escreva("Valor total em estoque: ", total)
	}
}
