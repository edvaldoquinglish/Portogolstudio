programa
{
	funcao inicio()
	{
		real saldo = 10000.0
		real valor
		inteiro opcao = 0

		enquanto(opcao != 4)
		{
			escreva("\n=== SISTEMA BANCÁRIO ===\n")
			escreva("1 - Ver saldo\n")
			escreva("2 - Depositar\n")
			escreva("3 - Sacar\n")
			escreva("4 - Sair\n")
			escreva("Escolha uma opção: ")

			leia(opcao)

			escolha(opcao)
			{
				caso 1:
					escreva("\nSaldo atual: ", saldo, "\n")
					pare

				caso 2:
					escreva("\nDigite valor do depósito: ")
					leia(valor)

					saldo = saldo + valor

					escreva("Depósito realizado!\n")
					pare

				caso 3:
					escreva("\nDigite valor do saque: ")
					leia(valor)

					se(valor <= saldo)
					{
						saldo = saldo - valor
						escreva("Saque realizado!\n")
					}
					senao
					{
						escreva("Saldo insuficiente!\n")
					}
					pare

				caso 4:
					escreva("Encerrando sistema...\n")
					pare

				caso contrario:
					escreva("Opção inválida!\n")
			}
		}
	}
}
